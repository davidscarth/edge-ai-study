vLLM:
```shell
sudo docker run --rm -it \
  --gpus all \
  --ipc=host \
  --network host \
  -v ~/.cache/huggingface:/root/.cache/huggingface \
  -e VLLM_USE_V2_MODEL_RUNNER=1 \
  ghcr.io/timothystewart6/vllm-gb10:latest \
  vllm serve RedHatAI/gemma-4-26B-A4B-it-NVFP4 \
  --host 0.0.0.0 --port 8000 \
  --gpu-memory-utilization 0.25 \
  --speculative-config '{"method": "mtp", "model": "google/gemma-4-26B-A4B-it-assistant", "num_speculative_tokens": 3}' \
  --kv-cache-dtype fp8 \
  --max-model-len 32768 \
  --max-num-seqs 6 \
  --max-num-batched-tokens 8192 \
  --enable-auto-tool-choice \
  --tool-call-parser gemma4 \
  --reasoning-parser gemma4 \
  --api-key "REPLACE-THIS-WITH-YOUR-OWN-KEY"
```
